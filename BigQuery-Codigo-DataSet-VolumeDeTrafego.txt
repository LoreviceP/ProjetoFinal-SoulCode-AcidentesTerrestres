CÓDIGO DA COSULTA FEITA EM SQL NO BIGQUERY NA GCP(GOOGLE CLOUD PLATFORM)


-- CONSULTA DA SOMA DO VOLUME TOTAL DE TRÁFEGO NAS PRAÇAS DE PEDÁGIO AGRUPADOS E ORDENADOS POR DATA
SELECT Data, SUM( Volume_total ) AS voltotal_201920 FROM `projeto-final-grupo03.grupo03.pedagio` GROUP BY Data ORDER BY Data;

-- CRIAR TABELA DA SOMA DO VOLUME TOTAL DE TRÁFEGO NAS PRAÇAS DE PEDÁGIO AGRUPADOS E ORDENADOS POR DATA
CREATE OR REPLACE TABLE grupo03.pedagio2020_pordia AS (SELECT DISTINCT Data, SUM( Volume_total ) AS voltotal_2020 FROM `projeto-final-grupo03.grupo03.pedagio` GROUP BY Data ORDER BY Data);
